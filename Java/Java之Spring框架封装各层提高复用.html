一: Dao层

1.新建一个封装包，从dao层开始，在该包下写一个BaseDao接口，里面向上抽取最常用的方法，例如增删改查。

package com.linyi.fengzhuang;

import java.io.Serializable;
import java.util.List;

public interface BaseDao<T> {
    //增
    public Serializable save(T t);
    public int save(List<T> lists);//返回成功了几个
    //删
    public void delete(Serializable id);
    //改
    public void update(T t);
    //查
    public T find(Serializable id);
    public List<T> find();
}

2.新建一个抽象类去实现BaseDao接口

package com.linyi.fengzhuang;

import org.hibernate.SessionFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.transaction.annotation.Transactional;

import java.io.Serializable;
import java.lang.reflect.ParameterizedType;
import java.util.List;

public abstract class BaseDaoImpl<T> implements BaseDao<T> {
    SessionFactory sessionFactory;
    Class<T> tClass;
    String className;

    @Autowired
    public void setSessionFactory(SessionFactory sessionFactory) {
        this.sessionFactory = sessionFactory;
    }

    public BaseDaoImpl(){
        //查找泛型的真正类型
        ParameterizedType genericSuperclass = (ParameterizedType) this.getClass().getGenericSuperclass();
        tClass = (Class<T>) genericSuperclass.getActualTypeArguments()[0];
        this.className = tClass.getSimpleName();
    }

//    @Transactional
    public Serializable save(T t) {
        Serializable serializable = sessionFactory.getCurrentSession().save(t);
//        sessionFactory.getCurrentSession().flush();
        return serializable;
    }

    public int save(List<T> lists){
        int i = 0;
        for(T t:lists){
            save(t);
            i++;
        }
        return i;
    }

    public void delete(Serializable id){
        T en = find(id);
        sessionFactory.getCurrentSession().delete(en);
    }

    public void update(T t){
        sessionFactory.getCurrentSession().update(t);
    }

    public T find(Serializable id){
        return (T) sessionFactory.getCurrentSession().get(tClass,id);
    }

    //查询全部

    public List<T> find() {
        String hql = "FROM "+this.className;
        return sessionFactory.getCurrentSession().createQuery(hql).list();
    }
}

二: Service层

1.在专门放封装类的接口的包下新建BaseService接口

package com.linyi.fengzhuang;

import java.io.Serializable;

public interface BaseService<T> {

    public Serializable save(T t);
    public T find(Serializable id);

}

2.再新建抽象类去实现

package com.linyi.fengzhuang;


import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.transaction.annotation.Transactional;

import java.io.Serializable;

public abstract class BaseServiceImpl<T> implements BaseService<T> {

    @Autowired
    private BaseDao<T> baseDao;

    @Transactional
    public Serializable save(T t) {
        return baseDao.save(t);
    }

    @Transactional
    public T find(Serializable id){
        return baseDao.find(id);
    }


    public void setBaseDao(BaseDao<T> baseDao) {
        this.baseDao = baseDao;
    }
}
